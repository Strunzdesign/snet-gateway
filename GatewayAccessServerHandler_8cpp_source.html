88  if (l_ServiceMessage.Deserialize(a_HigherLayerPayload)) {
89  // Check if it is directed to the address service

93  m_GatewayAccessProtocol.Send(l_AddressAssignmentReply.Serialize());
94  } // if

96  return;

102  if (l_PublishSubscribeConfirmation.GetDstServiceId() == 0xB0) {
103  m_GatewayAccessProtocol.Send(l_PublishSubscribeConfirmation.Serialize());
104  } // if
105
